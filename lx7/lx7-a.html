js

//获取所有元素'kp'
const kp = document.querySelectorAll('.h .kp');
//获取半径
const r = document.querySelector('.h').clientWidth / 2;
//读取元素数量
const sl = kp.length;
//获取每一个元素旋转的角度
const jd = 360 / sl;
for (let i = 0; i < sl; i++) {
let t=i * jd;
//转换弧度制
t=(Math.PI / 180) * t;
//利用三角函数计算
const x=Math.sin(t) * r,
y=-Math.cos(t) * r;
//进行平移变换
//``此处为反引号，引用ES6的写法，允许创建包含表达式和字符串
kp[i].style.transform=`translate(${x}px, ${y}px)`;
}
