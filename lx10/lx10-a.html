// 随机生成动画时长，在2~10秒之间
const randomDuration = Math.random() * 2 + 8;
// 获取骰子1，并设置动画时长
document.querySelector(".dice1").style.animationDuration = `${
randomDuration
}s, 7s`;
// 获取骰子3，并设置动画时长
document.querySelector(".dice3").style.animationDuration = `${
randomDuration - 1.5
}s, 7s`;
// 在运行1.5秒后显示骰子2
setTimeout(() => {
document.querySelector('.dice2').style.opacity = 1;
}, 1500);
